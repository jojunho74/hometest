🗄️ Supabase 테이블 생성 SQL

Supabase → SQL Editor에서 아래 SQL을 실행하세요 (최초 1회).

CREATE TABLE IF NOT EXISTS alio_source_cache (
  id            UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  url           TEXT NOT NULL,
  html_source   TEXT,
  dataset_name  TEXT,
  upload_data   JSONB,
  status        TEXT DEFAULT 'draft',
  created_at    TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

ALTER TABLE alio_source_cache ENABLE ROW LEVEL SECURITY;

CREATE POLICY "anon read" ON alio_source_cache FOR SELECT USING (true);
CREATE POLICY "auth insert" ON alio_source_cache FOR INSERT WITH CHECK (true);
CREATE POLICY "auth update" ON alio_source_cache FOR UPDATE USING (true);
CREATE POLICY "auth delete" ON alio_source_cache FOR DELETE USING (true);
