합격수기 수정

Supabase 테이블 생성 SQL

Supabase → SQL Editor에서 아래 SQL을 실행하세요:

CREATE TABLE success_stories (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  title TEXT,
  author_profile TEXT NOT NULL,
  target_score TEXT NOT NULL,
  prep_period TEXT NOT NULL,
  content TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

ALTER TABLE success_stories ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Anyone can read" ON success_stories
  FOR SELECT USING (true);

CREATE POLICY "Service role only insert" ON success_stories
  FOR INSERT WITH CHECK (true);
